120  template <typename ObjectDerivative, typename ObjectScriptMethodDerivative>
121  bool registerMethod()
122  {
123  std::string className = ObjectDerivative::classKeywordStatic();
124  std::string methodName = ObjectScriptMethodDerivative::classKeywordStatic();
125 
126  auto classEntryIt = m_factoryMap.find( className );
127  if ( classEntryIt != m_factoryMap.end() )
128  {
129  auto methodEntryIt = classEntryIt->second.find( methodName );
130  if ( methodEntryIt != classEntryIt->second.end() )
131  {
132  CAF_ASSERT( methodName != methodEntryIt->first ); // classNameKeyword has already been used
133  CAF_ASSERT( false ); // To be sure ..
134  return false; // never hit;
135  }
136  }
137  m_factoryMap[className][methodName] =
138  std::shared_ptr<ObjectMethodCreatorBase>( new ObjectMethodCreator<ObjectScriptMethodDerivative>() );
139  return true;
140  }
