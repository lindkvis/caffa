109  template <typename T>
110  void objectsByTypeStrict( std::vector<T*>* typedObjects, int selectionLevel = 0 )
111  {
112  std::vector<UiItem*> items;
113  this->selectedItems( items, selectionLevel );
114  for ( size_t i = 0; i < items.size(); i++ )
115  {
116  T* obj = dynamic_cast<T*>( items[i] );
117  if ( !obj )
118  {
119  typedObjects->clear();
120  break;
121  }
122  typedObjects->push_back( obj );
123  }
124  }
