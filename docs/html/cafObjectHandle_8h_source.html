243 template <typename T>
244 std::list<T*> ObjectHandle::descendantsOfType() const
245 {
246  std::list<T*> descendants;
247  for ( auto f : m_fields )
248  {
249  std::vector<ObjectHandle*> childObjects;
250  f->childObjects( &childObjects );
251 
252  for ( auto childObject : childObjects )
253  {
254  if ( childObject )
255  {
256  if ( T* typedObject = dynamic_cast<T*>( childObject ); typedObject )
257  {
258  descendants.push_back( typedObject );
259  }
260  std::list<T*> childDescendants = childObject->descendantsOfType<T>();
261  descendants.insert( descendants.end(), childDescendants.begin(), childDescendants.end() );
262  }
263  }
264  }
265  return descendants;
266 }
