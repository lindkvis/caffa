59 template <typename T>
60 bool findOptionItemValues( const std::deque<OptionItemInfo>& optionList, Variant fieldValue, std::vector<int>& foundIndexes )
61 {
62  foundIndexes.clear();
63 
64  // Find this fieldvalue in the optionlist if present
65 
66  // First handle lists/arrays of values
67  if ( fieldValue.isVector() )
68  {
69  std::vector<Variant> valuesSelectedInField = fieldValue.toVector();
70 
71  if ( valuesSelectedInField.size() )
72  {
73  // Create a list to be able to remove items as they are matched with values
74  std::vector<std::pair<Variant, int>> optionVariantAndIndexPairs;
75 
76  for ( int i = 0; i < optionList.size(); ++i )
77  {
78  Variant optionVariant( optionList[i].value() );
79  optionVariantAndIndexPairs.push_back( std::make_pair( optionVariant, i ) );
80  }
81 
82  for ( int i = 0; i < valuesSelectedInField.size(); ++i )
83  {
84  for ( auto it = optionVariantAndIndexPairs.begin(); it != optionVariantAndIndexPairs.end(); ++it )
85  {
86  if ( UiFieldSpecialization<T>::isDataElementEqual( valuesSelectedInField[i], it->first ) )
87  {
88  foundIndexes.push_back( it->second );
89 
90  // Assuming that one option is referenced only once, the option is erased. Then break
91  // out of the inner loop, as this operation can be costly for fields with many options and many
92  // values
93 
94  optionVariantAndIndexPairs.erase( it );
95  break;
96  }
97  }
98  }
99  }
100 
101  return ( static_cast<size_t>( valuesSelectedInField.size() ) <= foundIndexes.size() );
102  }
103  else // Then handle single value fields
104  {
105  for ( int opIdx = 0; opIdx < static_cast<int>( optionList.size() ); ++opIdx )
106  {
107  if ( UiFieldSpecialization<T>::isDataElementEqual( optionList[opIdx].value(), fieldValue ) )
108  {
109  foundIndexes.push_back( opIdx );
110  break;
111  }
112  }
113  return ( foundIndexes.size() > (size_t)0 );
114  }
115 }
