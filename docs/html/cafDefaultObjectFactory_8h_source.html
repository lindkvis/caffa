64 
65  auto entryIt = m_factoryMap.find( classNameKeyword );
66  if ( entryIt != m_factoryMap.end() )
67  {
68  CAF_ASSERT( classNameKeyword != entryIt->first ); // classNameKeyword has already been used
69  CAF_ASSERT( false ); // To be sure ..
70  return false; // never hit;
71  }
72  auto object = new ObjectCreator<ObjectBaseDerivative>();
73  m_factoryMap[classNameKeyword] = object;

75  }
